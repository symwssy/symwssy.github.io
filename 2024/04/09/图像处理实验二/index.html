  • # 读取图像
img = cv2.imread('E:/test_picture/IMG_2546.png')
# 显示原始图像和原始直方图
plt.figure(figsize=(10, 5))

plt.subplot(2, 2, 1)
plt.imshow(cv2.cvtColor(img, cv2.COLOR_BGR2RGB))
plt.title('Original Image')

# 转换为灰度图像
gray_img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
plt.subplot(2, 2, 2)
plt.hist(gray_img.ravel(), 256, [0, 256])
plt.title('Original Histogram')

# 取反图像
inverted_img = cv2.bitwise_not(gray_img)

plt.subplot(2, 2, 3)
plt.imshow(cv2.cvtColor(inverted_img, cv2.COLOR_GRAY2RGB))
plt.title('Inverted Image')

plt.subplot(2, 2, 4)
plt.hist(inverted_img.ravel(), 256, [0, 256])
plt.title('Inverted Histogram')

# ③ 直方图均衡
equalized_img = cv2.equalizeHist(gray_img)
plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.imshow(equalized_img, cmap='gray')
plt.title('Equalized Image')
# 查看均衡后的直方图
plt.subplot(1, 2, 2)
plt.hist(equalized_img.ravel(), 256, [0, 256])
plt.title('Equalized Histogram')

# ④ 旋转、切割，仅保留“爱丁堡花园”部分
# 旋转
rows, cols = gray_img.shape
M = cv2.getRotationMatrix2D((cols/2, rows/2), -10, 1)
rotated_img = cv2.warpAffine(gray_img, M, (cols, rows))
plt.figure(figsize=(10, 5))
plt.subplot(1, 2, 1)
plt.imshow(rotated_img, cmap='gray')
plt.title('Rotated Image')

# 切割出"爱丁堡花园"部分
crop_img = rotated_img[550:620, 600:900]

# 显示结果
plt.subplot(1, 2, 2)
plt.imshow(crop_img, cmap='gray')
plt.title('Cropped Image')

plt.show()
# 保存结果
cv2.imwrite('E:/test_picture/outputs/exp2_1_1.jpg', img.ravel())
cv2.imwrite('E:/test_picture/outputs/exp2_1_2.jpg', inverted_img.ravel())
cv2.imwrite('E:/test_picture/outputs/exp2_1_3.jpg', equalized_img.ravel())
cv2.imwrite('E:/test_picture/outputs/exp2_1_4.jpg', crop_img.ravel())

心得体会：

  • Sobel算子：用于边缘检测的离散微分算子。

当内核大小大于3时，可能产生比较明显的误差。

  • 在本实验中，我们将Sobel算子编码到PyTorch的卷积核中，通过对图像进行卷积操作，得到了图像的水平梯度、垂直梯度以及梯度幅值图像。这些梯度图像反映了图像中每个像素点的梯度变化情况，从而可以用于边缘检测、特征提取等图像处理任务。
